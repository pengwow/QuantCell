### 3. 策略类规范

必须继承 `Strategy`，遵循以下结构：

```python
class StrategyName(Strategy):
    """
    策略类（支持多品种）

    详细描述策略逻辑：
    - 核心指标计算方法
    - 入场条件
    - 出场条件
    - 风险控制

    Parameters
    ----------
    config : StrategyNameConfig
        策略配置对象
    """

    def __init__(self, config: StrategyNameConfig) -> None:
        super().__init__(config)
        # 使用 _config 存储配置，避免与父类的 config property 冲突
        self._config = config

        # 为每个品种维护独立的数据结构
        # key: instrument_id, value: 数据列表
        self.data: Dict[InstrumentId, List[float]] = {}

        # 当前指标值
        self.indicator1: Dict[InstrumentId, float] = {}
        self.indicator2: Dict[InstrumentId, float] = {}

        # 上一个指标值（用于判断交叉/突破）
        self.prev_indicator1: Dict[InstrumentId, float] = {}
        self.prev_indicator2: Dict[InstrumentId, float] = {}

        # 初始化每个品种的数据结构
        for instrument_id in config.instrument_ids:
            self.data[instrument_id] = []
            self.indicator1[instrument_id] = 0.0
            self.indicator2[instrument_id] = 0.0
            self.prev_indicator1[instrument_id] = 0.0
            self.prev_indicator2[instrument_id] = 0.0

    def on_start(self) -> None:
        """
        策略启动时调用
        """
        self.log_info(
            f"策略启动 - 参数1: {self._config.param1}, "
            f"参数2: {self._config.param2}, "
            f"品种数: {len(self._config.instrument_ids)}"
        )

    def on_bar(self, bar: Bar) -> None:
        """
        K线数据处理

        Parameters
        ----------
        bar : Bar
            K线数据对象，包含 open, high, low, close, volume 等属性
        """
        # 获取当前K线对应的品种ID
        instrument_id = bar.instrument_id

        # 保存当前指标值作为上一个值
        self.prev_indicator1[instrument_id] = self.indicator1[instrument_id]
        self.prev_indicator2[instrument_id] = self.indicator2[instrument_id]

        # 添加数据到历史
        close_price = bar.close
        self.data[instrument_id].append(close_price)

        # 保持历史数据长度（避免内存无限增长）
        max_period = max(self._config.param1, self._config.param2)
        if len(self.data[instrument_id]) > max_period * 2:
            self.data[instrument_id] = self.data[instrument_id][-max_period * 2:]

        # 计算指标
        if len(self.data[instrument_id]) >= self._config.param2:
            self.indicator1[instrument_id] = self._calculate_indicator1(
                self.data[instrument_id]
            )
            self.indicator2[instrument_id] = self._calculate_indicator2(
                self.data[instrument_id]
            )

            # 输出调试信息
            self.log_debug(
                f"[{instrument_id}] Close: {close_price:.2f}, "
                f"指标1: {self.indicator1[instrument_id]:.2f}, "
                f"指标2: {self.indicator2[instrument_id]:.2f}"
            )

            # 检查是否有足够的上一个值来判断信号
            if (
                self.prev_indicator1[instrument_id] > 0
                and self.prev_indicator2[instrument_id] > 0
            ):
                # 判断买入信号
                buy_signal = (
                    self.prev_indicator1[instrument_id]
                    <= self.prev_indicator2[instrument_id]
                    and self.indicator1[instrument_id] > self.indicator2[instrument_id]
                )

                # 判断卖出信号
                sell_signal = (
                    self.prev_indicator1[instrument_id]
                    >= self.prev_indicator2[instrument_id]
                    and self.indicator1[instrument_id] < self.indicator2[instrument_id]
                )

                # 执行交易逻辑
                if buy_signal:
                    self._on_buy_signal(bar, instrument_id)
                elif sell_signal:
                    self._on_sell_signal(bar, instrument_id)

    def _on_buy_signal(self, bar: Bar, instrument_id: InstrumentId) -> None:
        """
        买入信号处理

        Parameters
        ----------
        bar : Bar
            K线数据对象
        instrument_id : InstrumentId
            品种ID
        """
        self.log_info(
            f"[{instrument_id}] 买入信号！"
            f"指标1: {self.indicator1[instrument_id]:.2f} "
            f"上穿 指标2: {self.indicator2[instrument_id]:.2f}"
        )

        # 如果当前空仓，买入该品种
        if self.is_flat(instrument_id):
            self.log_info(f"[{instrument_id}] 当前空仓，执行买入")
            self.buy(instrument_id, self._config.trade_size)
        # 如果当前空头，先平仓再买入
        elif self.is_short(instrument_id):
            self.log_info(f"[{instrument_id}] 当前空头，先平仓再买入")
            self.close_position(instrument_id)
            self.buy(instrument_id, self._config.trade_size)
        else:
            self.log_info(f"[{instrument_id}] 当前已持有多头，无需操作")

    def _on_sell_signal(self, bar: Bar, instrument_id: InstrumentId) -> None:
        """
        卖出信号处理

        Parameters
        ----------
        bar : Bar
            K线数据对象
        instrument_id : InstrumentId
            品种ID
        """
        self.log_info(
            f"[{instrument_id}] 卖出信号！"
            f"指标1: {self.indicator1[instrument_id]:.2f} "
            f"下穿 指标2: {self.indicator2[instrument_id]:.2f}"
        )

        # 如果当前持有多头，卖出该品种
        if self.is_long(instrument_id):
            self.log_info(f"[{instrument_id}] 当前持有多头，执行卖出")
            self.sell(instrument_id, self._config.trade_size)
        else:
            self.log_info(f"[{instrument_id}] 当前未持有多头，无需操作")

    def on_stop(self) -> None:
        """
        策略停止时调用
        """
        self.log_info("策略停止")
        self.log_info(f"共处理 {self.bars_processed} 条K线数据")
```

## 关键要求

### 1. 多品种支持
- 必须为每个品种维护独立的数据结构（使用 Dict[InstrumentId, ...]）
- 在 `__init__` 中遍历 `config.instrument_ids` 初始化数据结构
- 在 `on_bar` 中通过 `bar.instrument_id` 获取当前品种

### 2. 信号判断逻辑
- 保存当前指标值到 `prev_*` 变量
- 计算新指标值
- 比较 `prev_*` 和当前值判断交叉/突破
- 避免在指标值不足时产生信号

### 3. 交易执行逻辑
- 使用 `self.is_flat()`, `self.is_long()`, `self.is_short()` 检查持仓状态
- 使用 `self.buy()`, `self.sell()`, `self.close_position()` 执行交易
- 使用 `self._config.trade_size` 作为交易数量

### 4. 日志记录
- 策略启动/停止时记录关键参数
- 信号产生时记录信号类型和指标值
- 交易执行时记录操作和原因
- 使用 `self.log_info()` 和 `self.log_debug()` 分级记录

### 5. 内存管理
- 限制历史数据长度（建议 `max_period * 2`）
- 避免无限增长的数据结构
