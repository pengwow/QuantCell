### 1. 向量化计算

向量化计算是量化策略中最重要的性能优化手段。

**优化前（低效）：**
```python
def calculate_sma(prices: list, window: int) -> list:
    """使用循环计算简单移动平均"""
    result = []
    for i in range(len(prices)):
        if i < window - 1:
            result.append(None)
        else:
            avg = sum(prices[i-window+1:i+1]) / window
            result.append(avg)
    return result
```

**优化后（高效）：**
```python
import numpy as np

def calculate_sma_vectorized(prices: np.ndarray, window: int) -> np.ndarray:
    """使用 NumPy 向量化计算简单移动平均"""
    weights = np.ones(window) / window
    return np.convolve(prices, weights, mode='valid')
```

**性能对比：**
- 循环版本：O(n×window) 时间复杂度
- 向量化版本：O(n) 时间复杂度，C语言级别优化

class StrategyCache:
    def __init__(self):
        self._indicator_cache = {}
        self._cache_hits = 0
        self._cache_misses = 0

    def get_or_calculate(self, key: str, calc_func, ttl: int = 300):
        """
        获取缓存值或计算新值

        Args:
            key: 缓存键
            calc_func: 计算函数
            ttl: 缓存有效期（秒）
        """
        import time
        current_time = time.time()

        if key in self._indicator_cache:
            value, timestamp = self._indicator_cache[key]
            if current_time - timestamp < ttl:
                self._cache_hits += 1
                return value

        # 缓存未命中，执行计算
        value = calc_func()
        self._indicator_cache[key] = (value, current_time)
        self._cache_misses += 1
        return value

    def clear_expired(self):
        """清理过期缓存"""
        import time
        current_time = time.time()
        expired_keys = [
            k for k, (_, ts) in self._indicator_cache.items()
            if current_time - ts > 300
        ]
        for k in expired_keys:
            del self._indicator_cache[k]
```

def find_signals_optimized(data):
    """使用向量化操作替代嵌套循环"""
    signals = []
    # 预计算条件
    mask = np.array([condition(d) for d in data])
    indices = np.where(mask)[0]

    # 使用 NumPy 的广播机制
    for idx in indices:
        # 向量化比较
        valid_j = indices[indices > idx]
        for j in valid_j:
            valid_k = indices[indices > j]
            signals.extend([(idx, j, k) for k in valid_k])

    return signals
```

def calculate_ma(data: list, period: int) -> float:
    """计算移动平均线"""
    if len(data) < period:
        return sum(data) / len(data)
    return sum(data[-period:]) / period

def calculate_rsi(data: list, period: int = 14) -> float:
    """计算RSI指标"""
    if len(data) < period + 1:
        return 50.0

    gains, losses = [], []
    for i in range(1, len(data)):
        change = data[i] - data[i-1]
        if change > 0:
            gains.append(change)
        else:
            losses.append(abs(change))

    avg_gain = sum(gains[-period:]) / period if gains else 0
    avg_loss = sum(losses[-period:]) / period if losses else 0

    if avg_loss == 0:
        return 100.0
    rs = avg_gain / avg_loss
    return 100 - (100 / (1 + rs))

def generate_signal(ma_fast: float, ma_mid: float, ma_slow: float, rsi: float) -> str:
    """根据技术指标生成交易信号"""
    is_uptrend = ma_fast > ma_mid > ma_slow
    is_downtrend = ma_fast < ma_mid < ma_slow

    if is_uptrend and rsi < 70:
        return 'buy'
    elif is_downtrend and rsi > 30:
        return 'sell'
    return 'hold'

def analyze_market_refactored(data: list) -> str:
    """重构后的市场分析函数"""
    ma5 = calculate_ma(data, 5)
    ma10 = calculate_ma(data, 10)
    ma20 = calculate_ma(data, 20)
    rsi = calculate_rsi(data)

    return generate_signal(ma5, ma10, ma20, rsi)

from typing import Protocol
from dataclasses import dataclass

class Strategy(Protocol):
    """策略接口"""
    def execute(self, data: dict) -> dict:
        ...

@dataclass
class StrategyRegistry:
    """策略注册表"""
    _strategies: dict = None

    def __post_init__(self):
        self._strategies = {}

    def register(self, name: str, strategy: Strategy):
        self._strategies[name] = strategy

    def get(self, name: str) -> Strategy:
        if name not in self._strategies:
            raise ValueError(f"Unknown strategy: {name}")
        return self._strategies[name]

# 使用
registry = StrategyRegistry()
registry.register('ma', MovingAverageStrategy())
registry.register('rsi', RSIStrategy())

def execute_strategy(strategy_type: str, data: dict) -> dict:
    strategy = registry.get(strategy_type)
    return strategy.execute(data)
